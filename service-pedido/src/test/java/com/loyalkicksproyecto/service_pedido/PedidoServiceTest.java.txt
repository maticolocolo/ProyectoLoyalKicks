package com.loyalkicksproyecto.service_pedido.service;

import static org.junit.jupiter.api.Assertions.*;
import static org.mockito.Mockito.*;

import java.time.LocalDateTime;
import java.util.Arrays;
import java.util.Collections;
import java.util.Optional;

import com.loyalkicksproyecto.service_pedido.model.DetallePedido;
import com.loyalkicksproyecto.service_pedido.model.Pedido;
import com.loyalkicksproyecto.service_pedido.repository.DetallePedidoRepository;
import com.loyalkicksproyecto.service_pedido.repository.PedidoRepository;

import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;
import org.junit.jupiter.api.extension.ExtendWith;

import org.mockito.InjectMocks;
import org.mockito.Mock;
import org.mockito.junit.jupiter.MockitoExtension;
import org.springframework.test.util.ReflectionTestUtils;
import org.springframework.web.reactive.function.client.WebClient;

@ExtendWith(MockitoExtension.class)
class PedidoServiceTest {

    @Mock
    private PedidoRepository pedidoRepository;

    @Mock
    private DetallePedidoRepository detallePedidoRepository;

    @Mock
    private WebClient.Builder webClientBuilder;

    @InjectMocks
    private PedidoService pedidoService;

    private Pedido pedido;

    @BeforeEach
    void setUp() {

        ReflectionTestUtils.setField(
                pedidoService,
                "detallePedidoRepository",
                detallePedidoRepository);

        ReflectionTestUtils.setField(
                pedidoService,
                "webClientBuilder",
                webClientBuilder);

        pedido = new Pedido();
        pedido.setId(1L);
        pedido.setUsuarioId(5L);
        pedido.setFecha(LocalDateTime.now());
        pedido.setEstado("Pendiente");
        pedido.setTotal(120000.0);
    }

    @Test
    void crearPedido_GuardarPedido() {

        // Given
        when(pedidoRepository.save(any(Pedido.class))).thenReturn(pedido);

        // When
        Pedido resultado = pedidoService.crearPedido(new Pedido());

        // Then
        assertNotNull(resultado);
        assertEquals("Pendiente", resultado.getEstado());

        verify(pedidoRepository).save(any(Pedido.class));
    }

    @Test
    void listarTodo_RetornaLista() {

        // Given
        when(pedidoRepository.findAll())
                .thenReturn(Arrays.asList(pedido));

        // When
        var lista = pedidoService.listarTodo();

        // Then
        assertEquals(1, lista.size());

        verify(pedidoRepository).findAll();
    }

    @Test
    void listarTodo_ListaVacia() {

        when(pedidoRepository.findAll())
                .thenReturn(Collections.emptyList());

        var lista = pedidoService.listarTodo();

        assertTrue(lista.isEmpty());

        verify(pedidoRepository).findAll();
    }

    @Test
    void buscarPorId_ExistePedido() {

        when(pedidoRepository.findById(1L))
                .thenReturn(Optional.of(pedido));

        Optional<Pedido> resultado = pedidoService.buscarPorId(1L);

        assertTrue(resultado.isPresent());
        assertEquals(1L, resultado.get().getId());

        verify(pedidoRepository).findById(1L);
    }

    @Test
    void buscarPorId_NoExistePedido() {

        when(pedidoRepository.findById(10L))
                .thenReturn(Optional.empty());

        Optional<Pedido> resultado = pedidoService.buscarPorId(10L);

        assertFalse(resultado.isPresent());

        verify(pedidoRepository).findById(10L);
    }

    @Test
    void cambiarEstado_Correctamente() {

        when(pedidoRepository.findById(1L))
                .thenReturn(Optional.of(pedido));

        when(pedidoRepository.save(any(Pedido.class)))
                .thenReturn(pedido);

        Pedido resultado = pedidoService.cambiarEstado(1L, "Confirmado");

        assertEquals("Confirmado", resultado.getEstado());

        verify(pedidoRepository).findById(1L);
        verify(pedidoRepository).save(any(Pedido.class));
    }

    @Test
    void cambiarEstado_LanzaExcepcion() {

        when(pedidoRepository.findById(5L))
                .thenReturn(Optional.empty());

        assertThrows(RuntimeException.class,
                () -> pedidoService.cambiarEstado(5L, "Cancelado"));

        verify(pedidoRepository).findById(5L);
    }

    @Test
    void eliminarPedido() {

        doNothing().when(pedidoRepository).deleteById(1L);

        pedidoService.eliminarPedido(1L);

        verify(pedidoRepository).deleteById(1L);
    }

    @Test
    void obtenerDetallePedido() {

        DetallePedido detalle = new DetallePedido();

        when(detallePedidoRepository.findByPedidoId(1L))
                .thenReturn(Arrays.asList(detalle));

        var lista = pedidoService.obtenerDetallePedido(1L);

        assertEquals(1, lista.size());

        verify(detallePedidoRepository).findByPedidoId(1L);
    }

}
