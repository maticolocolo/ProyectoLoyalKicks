package com.loyalkicksproyect.service_inventario.service;

import static org.junit.jupiter.api.Assertions.*;
import static org.mockito.Mockito.*;

import java.util.Arrays;
import java.util.Collections;
import java.util.Optional;

import com.loyalkicksproyect.service_inventario.model.Inventario;
import com.loyalkicksproyect.service_inventario.model.Talla;
import com.loyalkicksproyect.service_inventario.repository.InventarioRepository;
import com.loyalkicksproyect.service_inventario.repository.TallaRepository;

import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;
import org.junit.jupiter.api.extension.ExtendWith;

import org.mockito.InjectMocks;
import org.mockito.Mock;
import org.mockito.junit.jupiter.MockitoExtension;

import org.springframework.web.reactive.function.client.WebClient;

@ExtendWith(MockitoExtension.class)
class InventarioServiceTest {

    @Mock
    private InventarioRepository inventarioRepository;

    @Mock
    private TallaRepository tallaRepository;

    @Mock
    private WebClient.Builder webClientBuilder;

    @InjectMocks
    private InventarioService inventarioService;

    private Inventario inventario;
    private Talla talla;

    @BeforeEach
    void setUp() {

        talla = new Talla();
        talla.setId(1L);
        talla.setNombre("42");

        inventario = new Inventario();
        inventario.setId(1L);
        inventario.setProductoId(10L);
        inventario.setStock(15);
        inventario.setTalla(talla);
    }

    @Test
    void guardarTalla_DebeGuardarCorrectamente() {

        // Given
        when(tallaRepository.save(talla)).thenReturn(talla);

        // When
        Talla resultado = inventarioService.guardarTalla(talla);

        // Then
        assertNotNull(resultado);
        assertEquals(1L, resultado.getId());

        verify(tallaRepository).save(talla);
    }

    @Test
    void listarTallas_DebeRetornarLista() {

        when(tallaRepository.findAll())
                .thenReturn(Arrays.asList(talla));

        var lista = inventarioService.listarTallas();

        assertEquals(1, lista.size());

        verify(tallaRepository).findAll();
    }

    @Test
    void listarTallas_ListaVacia() {

        when(tallaRepository.findAll())
                .thenReturn(Collections.emptyList());

        var lista = inventarioService.listarTallas();

        assertTrue(lista.isEmpty());

        verify(tallaRepository).findAll();
    }

    @Test
    void buscarPorTId_Existe() {

        when(tallaRepository.findById(1L))
                .thenReturn(Optional.of(talla));

        Optional<Talla> resultado = inventarioService.buscarPorTId(1L);

        assertTrue(resultado.isPresent());
        assertEquals(1L, resultado.get().getId());

        verify(tallaRepository).findById(1L);
    }

    @Test
    void buscarPorTId_NoExiste() {

        when(tallaRepository.findById(10L))
                .thenReturn(Optional.empty());

        Optional<Talla> resultado = inventarioService.buscarPorTId(10L);

        assertFalse(resultado.isPresent());

        verify(tallaRepository).findById(10L);
    }

    @Test
    void deleteTalla_DebeEliminar() {

        doNothing().when(tallaRepository).deleteById(1L);

        inventarioService.deleteTalla(1L);

        verify(tallaRepository).deleteById(1L);
    }

    @Test
    void listarTodo_DebeRetornarInventario() {

        when(inventarioRepository.findAll())
                .thenReturn(Arrays.asList(inventario));

        var lista = inventarioService.listarTodo();

        assertEquals(1, lista.size());

        verify(inventarioRepository).findAll();
    }

    @Test
    void listarTodo_ListaVacia() {

        when(inventarioRepository.findAll())
                .thenReturn(Collections.emptyList());

        var lista = inventarioService.listarTodo();

        assertTrue(lista.isEmpty());

        verify(inventarioRepository).findAll();
    }

    @Test
    void listarPorProducto_DebeRetornarInventario() {

        when(inventarioRepository.findByProductoId(10L))
                .thenReturn(Arrays.asList(inventario));

        var lista = inventarioService.listarPorProducto(10L);

        assertEquals(1, lista.size());
        assertEquals(10L, lista.get(0).getProductoId());

        verify(inventarioRepository).findByProductoId(10L);
    }

    @Test
    void listarPorProducto_ListaVacia() {

        when(inventarioRepository.findByProductoId(20L))
                .thenReturn(Collections.emptyList());

        var lista = inventarioService.listarPorProducto(20L);

        assertTrue(lista.isEmpty());

        verify(inventarioRepository).findByProductoId(20L);
    }
}
