package com.loyalkicksproyect.sevice_producto.service;

import static org.junit.jupiter.api.Assertions.*;
import static org.mockito.Mockito.*;

import java.util.Arrays;
import java.util.List;
import java.util.Optional;

import com.loyalkicksproyect.sevice_producto.model.Producto;
import com.loyalkicksproyect.sevice_producto.repository.ProductoRespository;

import org.junit.jupiter.api.Test;
import org.junit.jupiter.api.extension.ExtendWith;
import org.mockito.InjectMocks;
import org.mockito.Mock;
import org.mockito.junit.jupiter.MockitoExtension;

@ExtendWith(MockitoExtension.class)
class ProductoServiceTest {

    @Mock
    private ProductoRespository productoRespository;

    @InjectMocks
    private ProductoService productoService;

    @Test
    void listarTodoPTest() {

        Producto p1 = new Producto();
        Producto p2 = new Producto();

        when(productoRespository.findAll())
                .thenReturn(Arrays.asList(p1, p2));

        List<Producto> lista = productoService.listarTodoP();

        assertEquals(2, lista.size());

        verify(productoRespository, times(1)).findAll();
    }

    @Test
    void buscarPorIdPTest() {

        Producto producto = new Producto();
        producto.setId_producto(1L);

        when(productoRespository.findById(1L))
                .thenReturn(Optional.of(producto));

        Optional<Producto> resultado = productoService.buscarPorIdP(1L);

        assertTrue(resultado.isPresent());
        assertEquals(1L, resultado.get().getId_producto());

        verify(productoRespository).findById(1L);
    }

    @Test
    void guardarProductoTest() {

        Producto producto = new Producto();
        producto.setNombre_producto("Nike Air Max");

        when(productoRespository.save(any(Producto.class)))
                .thenReturn(producto);

        Producto resultado = productoService.guardaProducto(producto);

        assertNotNull(resultado);
        assertEquals("Nike Air Max", resultado.getNombre_producto());

        verify(productoRespository).save(producto);
    }

    @Test
    void eliminarProductoTest() {

        doNothing().when(productoRespository).deleteById(1L);

        productoService.eliminarProducto(1L);

        verify(productoRespository, times(1)).deleteById(1L);
    }

}
