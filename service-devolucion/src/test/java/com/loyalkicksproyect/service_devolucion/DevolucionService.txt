package com.loyalkicksproyect.service_devolucion.service;

import com.loyalkicksproyect.service_devolucion.model.Devolucion;
import com.loyalkicksproyect.service_devolucion.repository.DevolucionRepository;
import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;
import org.junit.jupiter.api.extension.ExtendWith;

import org.mockito.InjectMocks;
import org.mockito.Mock;
import org.mockito.junit.jupiter.MockitoExtension;

import org.springframework.web.reactive.function.client.WebClient;
import reactor.core.publisher.Mono;

import java.time.LocalDate;
import java.util.Arrays;
import java.util.Optional;

import static org.junit.jupiter.api.Assertions.*;
import static org.mockito.ArgumentMatchers.*;
import static org.mockito.Mockito.*;

@ExtendWith(MockitoExtension.class)
class DevolucionServiceTest {

    @Mock
    private DevolucionRepository devolucionRepository;

    @Mock
    private WebClient.Builder webClientBuilder;

    @Mock
    private WebClient webClient;

    @Mock
    private WebClient.RequestHeadersUriSpec requestHeadersUriSpec;

    @Mock
    private WebClient.RequestHeadersSpec requestHeadersSpec;

    @Mock
    private WebClient.ResponseSpec responseSpec;

    @InjectMocks
    private DevolucionService devolucionService;

    private Devolucion devolucion;

    @BeforeEach
    void setUp() {

        devolucion = new Devolucion();

        devolucion.setId(1L);
        devolucion.setPedidoId(5L);
        devolucion.setProductoId(10L);
        devolucion.setMotivo("Producto defectuoso");

    }

    private void mockWebClient() {

        when(webClientBuilder.build()).thenReturn(webClient);

        when(webClient.get()).thenReturn(requestHeadersUriSpec);

        when(requestHeadersUriSpec.uri(anyString()))
                .thenReturn(requestHeadersSpec);

        when(requestHeadersSpec.retrieve())
                .thenReturn(responseSpec);

        when(responseSpec.bodyToMono(Object.class))
                .thenReturn(Mono.just("Pedido"));
    }

    @Test
    void solicitarDevolucion_DebeGuardarCorrectamente() {

        mockWebClient();

        when(devolucionRepository.save(any()))
                .thenAnswer(i -> i.getArgument(0));

        Devolucion resultado =
                devolucionService.solicitarDevolucion(devolucion);

        assertNotNull(resultado);

        assertEquals("PENDIENTE", resultado.getEstado());

        assertEquals(LocalDate.now(), resultado.getFechaSolicitud());

        verify(devolucionRepository).save(any());

    }

    @Test
    void listarTodo_DebeRetornarLista() {

        when(devolucionRepository.findAll())
                .thenReturn(Arrays.asList(devolucion));

        assertEquals(1, devolucionService.listarTodo().size());

        verify(devolucionRepository).findAll();

    }

    @Test
    void buscarPorId_DebeRetornarDevolucion() {

        mockWebClient();

        when(devolucionRepository.findById(1L))
                .thenReturn(Optional.of(devolucion));

        Optional<Devolucion> resultado =
                devolucionService.buscarPorId(1L);

        assertTrue(resultado.isPresent());

        assertEquals(1L, resultado.get().getId());

    }

    @Test
    void buscarPorId_NoExiste() {

        when(devolucionRepository.findById(1L))
                .thenReturn(Optional.empty());

        Optional<Devolucion> resultado =
                devolucionService.buscarPorId(1L);

        assertTrue(resultado.isEmpty());

    }

    @Test
    void cambiarEstado_DebeActualizarEstado() {

        when(devolucionRepository.findById(1L))
                .thenReturn(Optional.of(devolucion));

        when(devolucionRepository.save(any()))
                .thenAnswer(i -> i.getArgument(0));

        Devolucion actualizado =
                devolucionService.cambiarEstado(1L,"APROBADA");

        assertEquals("APROBADA", actualizado.getEstado());

        verify(devolucionRepository).save(any());

    }

    @Test
    void cambiarEstado_NoExiste_DebeLanzarExcepcion() {

        when(devolucionRepository.findById(1L))
                .thenReturn(Optional.empty());

        assertThrows(RuntimeException.class,
                () -> devolucionService.cambiarEstado(1L,"RECHAZADA"));

    }

}
