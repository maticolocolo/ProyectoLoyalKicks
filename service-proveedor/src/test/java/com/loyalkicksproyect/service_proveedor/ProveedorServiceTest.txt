package com.loyalkicksproyect.service_proveedor.service;

import static org.junit.jupiter.api.Assertions.*;
import static org.mockito.Mockito.*;

import java.util.Arrays;
import java.util.Collections;
import java.util.Optional;

import com.loyalkicksproyect.service_proveedor.model.ProductoProveedor;
import com.loyalkicksproyect.service_proveedor.model.Proveedor;
import com.loyalkicksproyect.service_proveedor.repository.ProductoProveedorRepository;
import com.loyalkicksproyect.service_proveedor.repository.ProveedorRepository;

import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;
import org.junit.jupiter.api.extension.ExtendWith;

import org.mockito.InjectMocks;
import org.mockito.Mock;
import org.mockito.junit.jupiter.MockitoExtension;

import org.springframework.test.util.ReflectionTestUtils;
import org.springframework.web.reactive.function.client.WebClient;

@ExtendWith(MockitoExtension.class)
class ProveedorServiceTest {

    @Mock
    private ProveedorRepository proveedorRepository;

    @Mock
    private ProductoProveedorRepository productoProveedorRepository;

    @Mock
    private WebClient.Builder webClientBuilder;

    @InjectMocks
    private ProveedorService proveedorService;

    private Proveedor proveedor;
    private ProductoProveedor relacion;

    @BeforeEach
    void setUp() {

        ReflectionTestUtils.setField(
                proveedorService,
                "webClientBuilder",
                webClientBuilder);

        proveedor = new Proveedor();
        proveedor.setId(1L);
        proveedor.setRazonSocial("Nike");
        proveedor.setRut("11.111.111-1");
        proveedor.setContactoNombre("Juan");
        proveedor.setContactoTelefono("999999999");
        proveedor.setContactoCorreo("nike@test.cl");

        relacion = new ProductoProveedor();
        relacion.setId(1L);
        relacion.setProductoId(5L);
        relacion.setProveedor(proveedor);
    }

    @Test
    void agregarProveedor_DebeGuardar() {

        when(proveedorRepository.save(proveedor))
                .thenReturn(proveedor);

        Proveedor resultado = proveedorService.agregarProveedor(proveedor);

        assertNotNull(resultado);
        assertEquals("Nike", resultado.getRazonSocial());

        verify(proveedorRepository).save(proveedor);
    }

    @Test
    void listarTodo_DebeRetornarLista() {

        when(proveedorRepository.findAll())
                .thenReturn(Arrays.asList(proveedor));

        var lista = proveedorService.listarTodo();

        assertEquals(1, lista.size());

        verify(proveedorRepository).findAll();
    }

    @Test
    void listarTodo_ListaVacia() {

        when(proveedorRepository.findAll())
                .thenReturn(Collections.emptyList());

        assertTrue(proveedorService.listarTodo().isEmpty());

        verify(proveedorRepository).findAll();
    }

    @Test
    void buscarPorId_Existe() {

        when(proveedorRepository.findById(1L))
                .thenReturn(Optional.of(proveedor));

        Optional<Proveedor> resultado = proveedorService.buscarPorId(1L);

        assertTrue(resultado.isPresent());
        assertEquals(1L, resultado.get().getId());

        verify(proveedorRepository).findById(1L);
    }

    @Test
    void buscarPorId_NoExiste() {

        when(proveedorRepository.findById(20L))
                .thenReturn(Optional.empty());

        assertFalse(proveedorService.buscarPorId(20L).isPresent());

        verify(proveedorRepository).findById(20L);
    }

    @Test
    void actualizarProveedor_DebeActualizarDatos() {

        Proveedor nuevos = new Proveedor();
        nuevos.setRazonSocial("Adidas");
        nuevos.setRut("22.222.222-2");
        nuevos.setContactoNombre("Pedro");
        nuevos.setContactoTelefono("888888888");
        nuevos.setContactoCorreo("adidas@test.cl");

        when(proveedorRepository.findById(1L))
                .thenReturn(Optional.of(proveedor));

        when(proveedorRepository.save(any(Proveedor.class)))
                .thenReturn(proveedor);

        Proveedor resultado =
                proveedorService.actualizarProveedor(1L, nuevos);

        assertEquals("Adidas", resultado.getRazonSocial());
        assertEquals("22.222.222-2", resultado.getRut());

        verify(proveedorRepository).findById(1L);
        verify(proveedorRepository).save(any(Proveedor.class));
    }

    @Test
    void actualizarProveedor_NoExiste() {

        when(proveedorRepository.findById(100L))
                .thenReturn(Optional.empty());

        assertThrows(RuntimeException.class,
                () -> proveedorService.actualizarProveedor(100L, new Proveedor()));

        verify(proveedorRepository).findById(100L);
    }

    @Test
    void eliminarProveedor() {

        doNothing().when(proveedorRepository).deleteById(1L);

        proveedorService.eliminarProveedor(1L);

        verify(proveedorRepository).deleteById(1L);
    }

    @Test
    void asociarProducto_DebeGuardarRelacion() {

        when(proveedorRepository.findById(1L))
                .thenReturn(Optional.of(proveedor));

        when(productoProveedorRepository.save(relacion))
                .thenReturn(relacion);

        ProductoProveedor resultado =
                proveedorService.asociarProducto(relacion);

        assertNotNull(resultado);
        assertEquals(5L, resultado.getProductoId());

        verify(proveedorRepository).findById(1L);
        verify(productoProveedorRepository).save(relacion);
    }

    @Test
    void asociarProducto_ProveedorNoExiste() {

        when(proveedorRepository.findById(1L))
                .thenReturn(Optional.empty());

        assertThrows(RuntimeException.class,
                () -> proveedorService.asociarProducto(relacion));

        verify(proveedorRepository).findById(1L);
    }
}
